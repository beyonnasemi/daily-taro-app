export const getRandomCard = () => {
  const randomIndex = Math.floor(Math.random() * tarotCards.length);
  const isReversed = Math.random() < 0.3; // 30% 확률로 역방향
  return {
    ...tarotCards[randomIndex],
    isReversed,
    timestamp: new Date().toISOString()
  };
};
